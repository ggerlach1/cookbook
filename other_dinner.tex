\begin{recipe}[source=NYT cooking, preparationtime= 25minutes]
		{Ramen Carbonara}
		\ingredients[7]{
				6-8            & slices of bacon \\
				\unit[2]{cups} & boiling water \\
				\unit[3]{tbsp} & butter \\
				\unit[1]{cup}  & grated Parmesan \\
				2              & packages ramen noodles \\
				2              & eggs, whisked \\
				               & kosher salt\\
				               & black pepper \\
		}

		\preparation{
				\step Slice bacon into 1/2-inch strips and cook over medium heat in a large skillet until crispy. Use a slotted spoon to transfer it to a paper-towel lined plate. Set aside.
				\step Pour off the extra bacon fat in the ban. Return the pan to medium-high heat and add boiling water butter and half of the cheese. Stir, scraping up bacon bits, and bring to a boil. Add noodle blocks and boil, gently separating the strands as they soften, until noodles are almost cooked through, about 3  minutes. The noodles will absorb some of the liquid, and there will be a thick broth in the pan. Keep the heat high; you want most of the liquid to evaporate.
				\step Reduce heat to low. Add whisked eggs stir into noodles very well, and cook, stirring constantly and scraping the bottom of the pan, about 1 minute. The sauce should remain quite tunny; the eggs will continue to cook after you remove them from the heat. Mix in bacon, remaining cheese and plenty of black pepper.
				\step Scrape mixture into 2 large serving bowls. Serve grinding pepper over the top of each bowl.
		}
\end{recipe}

\begin{recipe}[source = dad, preparationtime = 1 hour]{Risotto}
		\ingredients[7]{
				\unit[1]{cup}  & Arborio rice \\
				\unit[4]{cups} & Vegetable broth \\
				1              & onion, diced \\
				1              & ground Italian sausage (or fake sausage) \\
				\unit[3]{cups} & other vegetables, diced \\
							   & parmesan cheese \\
				}

		\preparation{
				\step Heat a small amount of neutral cooking oil in a large skillet. Cook the onion, sausage, and other vegetables until well sauteed and soft. Remove from pan.
				\step Add the rice to the hot pan and toast for several minutes on medium heat until it begins to become translucent.
				\step Add about 1/2 cup of the vegetable broth to the pan the stir the rice and liquid. Once nearly all the liquid has been absorbed, add another half cup and continue cooking. Continue this process until you have used nearly all the vegetable broth. Begin tasting the rice for being fully cooked after you have added about 3 cups of liquid.
				\step When rice is nearly done add the sausage and vegetables back in. Fully combine and allow to reheat.
				\step Serve hot topped with cheese, or store in air-tight container in the fridge for up to 5 days.
		}
\end{recipe}
