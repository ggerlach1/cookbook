\section{Food of Unclear Origins - potentially American}

\begin{recipe}[source=NYT cooking, preparationtime= 25minutes]
		{Ramen Carbonara}
		\ingredients[7]{
				6-8 & slices of bacon \\
				\unit[2]{cups} & boiling water \\
				\unit[3]{tbsp} & butter \\
				\unit[1]{cup} & grated Parmesan \\
				2 & packages ramen noodles \\
				2 & eggs, whisked \\
				  & kosher salt\\
				  & black pepper \\
		}

		\preparation{
				\step Slice bacon into 1/2-inch strips and cook over medium heat in a large skillet until crispy. Use a slotted spoon to transfer it to a paper-towel lined plate. Set aside.
				\step Pour off the extra bacon fat in the ban. Return the pan to medium-high heat and add boiling water butter and half of the cheese. Stir, scraping up bacon bits, and bring to a boil. Add noodle blocks and boil, gently separating the strands as they soften, until noodles are almost cooked through, about 3  minutes. The noodles will absorb some of the liquid, and there will be a thick broth in the pan. Keep the heat high; you want most of the liquid to evaporate.
				\step Reduce heat to low. Add whisked eggs stir into noodles very well, and cook, stirring constantly and scraping the bottom of the pan, about 1 minute. The sauce should remain quite tunny; the eggs will continue to cook after you remove them from the heat. Mix in bacon, remaining cheese and plenty of black pepper.
				\step Scrape mixture into 2 large serving bowls. Serve grinding pepper over the top of each bowl.
		}
\end{recipe}
